## ğŸš€ HOW TO USE

### 1. Add Routes to Server

Edit `backend/server.js` and add:

```javascript
// Import new routes
const razorpayRoutes = require('./routes/razorpay');
const mailchimpRoutes = require('./routes/mailchimp');
const paypalRoutes = require('./routes/paypal');
const telegramBotRoutes = require('./routes/telegram-bot');
const dropboxRoutes = require('./routes/dropbox-storage');
const asanaRoutes = require('./routes/asana-pm');
const zendeskRoutes = require('./routes/zendesk-support');
const calendlyRoutes = require('./routes/calendly-scheduling');
const googleAnalyticsRoutes = require('./routes/google-analytics');
const openaiRoutes = require('./routes/openai-gpt');
const cloudinaryRoutes = require('./routes/cloudinary-media');
const plaidRoutes = require('./routes/plaid-banking');
const resendRoutes = require('./routes/resend-email');
const redisRoutes = require('./routes/redis-cache');
const mongodbRoutes = require('./routes/mongodb-atlas');

// Mount routes
app.use('/api/razorpay', razorpayRoutes);
app.use('/api/mailchimp', mailchimpRoutes);
app.use('/api/paypal', paypalRoutes);
app.use('/api/telegram-bot', telegramBotRoutes);
app.use('/api/dropbox', dropboxRoutes);
app.use('/api/asana', asanaRoutes);
app.use('/api/zendesk', zendeskRoutes);
app.use('/api/calendly', calendlyRoutes);
app.use('/api/google-analytics', googleAnalyticsRoutes);
app.use('/api/openai', openaiRoutes);
app.use('/api/cloudinary', cloudinaryRoutes);
app.use('/api/plaid', plaidRoutes);
app.use('/api/resend', resendRoutes);
app.use('/api/redis', redisRoutes);
app.use('/api/mongodb', mongodbRoutes);
```

### 2. Install Dependencies

```bash
npm install razorpay @mailchimp/mailchimp_marketing openai cloudinary plaid resend ioredis mongodb node-telegram-bot-api dropbox asana axios
```

### 3. Configure Environment Variables

Create `.env` file:

```env
# Razorpay
RAZORPAY_KEY_ID=your_key_id
RAZORPAY_KEY_SECRET=your_key_secret

# Mailchimp
MAILCHIMP_API_KEY=your_api_key
MAILCHIMP_SERVER_PREFIX=us1

# PayPal
PAYPAL_CLIENT_ID=your_client_id
PAYPAL_CLIENT_SECRET=your_client_secret
PAYPAL_MODE=sandbox

# Telegram
TELEGRAM_BOT_TOKEN=your_bot_token

# Dropbox
DROPBOX_ACCESS_TOKEN=your_access_token

# Asana
ASANA_ACCESS_TOKEN=your_access_token

# Zendesk
ZENDESK_SUBDOMAIN=your_subdomain
ZENDESK_EMAIL=your_email
ZENDESK_API_TOKEN=your_api_token

# Calendly
CALENDLY_API_KEY=your_api_key

# Google Analytics
GA_PROPERTY_ID=your_property_id
GOOGLE_APPLICATION_CREDENTIALS=path/to/credentials.json

# OpenAI
OPENAI_API_KEY=your_api_key

# Cloudinary
CLOUDINARY_CLOUD_NAME=your_cloud_name
CLOUDINARY_API_KEY=your_api_key
CLOUDINARY_API_SECRET=your_api_secret

# Plaid
PLAID_CLIENT_ID=your_client_id
PLAID_SECRET=your_secret
PLAID_ENV=sandbox

# Resend
RESEND_API_KEY=your_api_key

# Redis
REDIS_HOST=localhost
REDIS_PORT=6379
REDIS_PASSWORD=your_password

# MongoDB
MONGODB_URI=mongodb+srv://username:password@cluster.mongodb.net/database
```
